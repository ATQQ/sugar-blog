## Usage
安装依赖 `pnpm/npm/yarn`
```sh
pnpm add vitepress-plugin-announcement
```

引入插件在 `.vitepress/config.ts` 配置文件中

```ts
import { defineConfig } from 'vitepress'

export default defineConfig({
  vite: {
    // ↓↓↓↓↓
    plugins: [

    ]
    // ↑↑↑↑↑
  }
})
```
